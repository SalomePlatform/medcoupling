  ${MED3_INCLUDE_DIRS}
  ${HDF5_INCLUDE_DIRS}

  ${CMAKE_CURRENT_SOURCE_DIR}
  ${CMAKE_CURRENT_SOURCE_DIR}/../MEDSPLITTER

  ${CMAKE_CURRENT_SOURCE_DIR}/../MEDMEM_SWIG
  ${CMAKE_CURRENT_SOURCE_DIR}/../INTERP_KERNEL
